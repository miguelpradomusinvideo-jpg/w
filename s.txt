 Comandos de texto
client.on('messageCreate', async (message) = {
    if (message.author.bot) return;
    if (!message.content.startsWith(PREFIX)) return;

    const args = message.content.slice(PREFIX.length).trim().split( +);
    const command = args.shift().toLowerCase();

    switch (command) {
        case 'add'
            await addAccountFormat(message, args);
            break;
        case 'gerar'
            await generateCode(message, args);
            break;
        case 'listar'
            await listAccounts(message);
            break;
        case 'remover'
            await removeAccount(message, args);
            break;
        case 'historico'
            await showHistory(message, args);
            break;
        case 'ajuda'
            await sendHelp(message);
            break;
        case 'creditos'
            await sendCredits(message);
            break;
        default
            break;
    }
});

 Slash commands
client.on('interactionCreate', async (interaction) = {
    if (!interaction.isCommand() && !interaction.isButton() && !interaction.isModalSubmit() && !interaction.isStringSelectMenu()) return;

    try {
        if (interaction.isCommand()) {
            const { commandName } = interaction;
            
            switch (commandName) {
                case 'add'
                    await showAddModal(interaction);
                    break;
                case 'gerar'
                    await showAccountsSelect(interaction);
                    break;
                case 'listar'
                    await listAccountsSlash(interaction);
                    break;
                case 'creditos'
                    await sendCreditsSlash(interaction);
                    break;
            }
        } else if (interaction.isButton()) {
            await handleButtonInteraction(interaction);
        } else if (interaction.isModalSubmit()) {
            await handleModalSubmit(interaction);
        } else if (interaction.isStringSelectMenu()) {
            await handleSelectMenu(interaction);

            .setColor(0xFF0000)
            .setTitle('❌ Erro')
            .setDescription('Ocorreu um erro ao processar sua solicitação.')
            .setFooter({ text 'Gg Community © Criado por Miguel®' });
        
        await interaction.reply({ embeds [embed], ephemeral true });
    }
});

 FUNÇÃO PRINCIPAL Adicionar conta no formato EMAIL  SENHA  2FA
async function addAccountFormat(message, args) {
    const input = args.join(' ');
    
     Verificar se está no formato EMAIL  SENHA  2FA
    const regex = ^([^]+)ss([^]+)ss([A-Z2-7]+)$i;
    const match = input.match(regex);
    
    if (!match) {
        const embed = new EmbedBuilder()
            .setColor(0xFF0000)
            .setTitle('❌ Formato Inválido!')
            .setDescription('Use o formato EMAIL  SENHA  2FAnnExemplon`!add miguel@gmail.com  minhaSenha123  JBSWY3DPEHPK3PXP`')
            .setFooter({ text 'Gg Community © Criado por Miguel®' });
        
        return message.reply({ embeds [embed] });
    }
    
    const email = match[1].trim();
    const senha = match[2].trim();
    const secret2FA = match[3].trim().toUpperCase();
    
     Validar chave 2FA
    const testCode = speakeasy.totp({
        secret secret2FA,
        encoding 'base32',
        step 30,
        digits 6
    });
    
    if (!testCode  testCode.length !== 6) {
        const embed = new EmbedBuilder()
            .setColor(0xFF0000)
            .setTitle('❌ Chave 2FA Inválida!')
            .setDescription('A chave 2FA deve estar no formato Base32.nExemplo `JBSWY3DPEHPK3PXP`')
            .setFooter({ text 'Gg Community © Criado por Miguel®' });
        
        return message.reply({ embeds [embed] });
    }
    
    try {
        database.saveAccount(message.author.id, email, senha, secret2FA);
        
        const embed = new EmbedBuilder()
            .setColor(0x00FF00)
            .setTitle('✅ Conta Rockstar Adicionada!')
            .setDescription(`Conta ${email} foi salva com sucesso!`)
            .addFields(
                { name '📧 Email', value email, inline true },
                { name '🔐 Senha', value '••••••••', inline true },
                { name '🔑 2FA', value '✅ Configurado', inline true },
                { name '🎮 Como usar', value 'Use `!gerar ' + email.split('@')[0] + '` ou `!gerar ' + email + '` para gerar o código 2FA', inline false }
            )
            .setFooter({ text 'Gg Community © Criado por Miguel®' })
            .setTimestamp();
        
        await message.reply({ embeds [embed] });
    } catch (error) {
        await message.reply('❌ Erro ao salvar conta. Tente novamente.');
    }
}

 FUNÇÃO PRINCIPAL Gerar código 2FA
async function generateCode(message, args) {
    const searchTerm = args.join(' ');
    
    if (!searchTerm) {
        const accounts = database.getUserAccounts(message.author.id);
        if (accounts.length === 0) {
            const embed = new EmbedBuilder()
                .setColor(0xFFA500)
                .setTitle('📭 Nenhuma conta cadastrada')
                .setDescription('Use o comando para adicionar uma conta no formaton`!add EMAIL  SENHA  2FA`nnExemplon`!add miguel@gmail.com  123456  JBSWY3DPEHPK3PXP`')
                .setFooter({ text 'Gg Community © Criado por Miguel®' });
            
            return message.reply({ embeds [embed] });
        }
        
         Mostrar lista de contas
        const embed = new EmbedBuilder()
            .setColor(0x00FF00)
            .setTitle('🎮 Suas Contas Rockstar')
            .setDescription('Use `!gerar email` para gerar o código 2FA')
            .addFields(
                accounts.map(acc = ({ 
                    name `📧 ${acc.email}`, 
                    value `🎮 Conta ${acc.account_name  'Sem nome'}n📅 Adicionada ${new Date(acc.created_at).toLocaleDateString()}`,
                    inline false 
                }))
            )
            .setFooter({ text 'Gg Community © Criado por Miguel®' });
        
        return message.reply({ embeds [embed] });
    }
    
     Buscar conta por email ou parte do email
    let account = database.getAccountByEmail(message.author.id, searchTerm);
    
    if (!account) {
         Buscar por parte do email
        const accounts = database.searchAccounts(message.author.id, searchTerm);
        if (accounts.length === 0) {
            return message.reply(`❌ Conta com email ${searchTerm} não encontrada. Use `!listar` para ver suas contas.`);
        }
        if (accounts.length  1) {
            const embed = new EmbedBuilder()
                .setColor(0xFFA500)
                .setTitle('🔍 Múltiplas contas encontradas')
                .setDescription('Use o email completo para gerar o código')
                .addFields(accounts.map(acc = ({ name acc.email, value 'Clique para copiar', inline false })))
                .setFooter({ text 'Gg Community © Criado por Miguel®' });
            
            return message.reply({ embeds [embed] });
        }
        account = accounts[0];
         Buscar dados completos
        account = database.getAccountById(account.id, message.author.id);
    }
    
     Gerar código TOTP
    const code = speakeasy.totp({
        secret account.secret_2fa,
        encoding 'base32',
        step 30,
        digits 6
    });
    
     Calcular tempo restante
    const now = Math.floor(Date.now()  1000);
    const timeStep = 30;
    const timeLeft = timeStep - (now % timeStep);
    
     Salvar no histórico
    database.logGeneratedCode(account.id, account.email, code);
    
     Armazenar código ativo
    ACTIVE_CODES.set(`${message.author.id}_${account.email}`, {
        code,
        expiresAt now + timeLeft,
        accountId account.id
    });
    
     Criar embed com informações completas
    const embed = new EmbedBuilder()
        .setColor(0x00FF00)
        .setTitle('🎮 ROCKSTAR 2FA - Código Gerado')
        .setDescription(`Conta ${account.email}`)
        .addFields(
            { name '🔐 CÓDIGO 2FA', value ````n${code}n````, inline false },
            { name '📧 Email', value account.email, inline true },
            { name '🔑 Senha', value '' + account.senha + '', inline true },
            { name '⏱️ Expira em', value `${timeLeft} segundos`, inline true },
            { name '🎮 Plataforma', value 'Rockstar Games (GTA V, RDR2)', inline false }
        )
        .setFooter({ text 'Use este código para fazer login na Rockstar Launcher  Gg Community © Criado por Miguel®' })
        .setTimestamp();
    
    const row = new ActionRowBuilder()
        .addComponents(
            new ButtonBuilder()
                .setCustomId(`copy_code_${code}_${account.email}`)
                .setLabel('📋 Copiar Código')
                .setStyle(ButtonStyle.Primary),
            new ButtonBuilder()
                .setCustomId(`refresh_code_${account.id}_${account.email}`)
                .setLabel('🔄 Gerar Novamente')
                .setStyle(ButtonStyle.Success),
            new ButtonBuilder()
                .setCustomId(`copy_creds_${account.id}_${account.email}`)
                .setLabel('📧 Copiar EmailSenha')
                .setStyle(ButtonStyle.Secondary)
        );
    
    const msg = await message.reply({ embeds [embed], components [row] });
    
     Timer para atualizar automaticamente
    setTimeout(async () = {
        const newCode = speakeasy.totp({
            secret account.secret_2fa,
            encoding 'base32',
            step 30,
            digits 6
        });
        
        const newTimeLeft = 30 - (Math.floor(Date.now()  1000) % 30);
        
        const updatedEmbed = new EmbedBuilder()
            .setColor(0x00FF00)
            .setTitle('🎮 ROCKSTAR 2FA - Código Gerado')
            .setDescription(`Conta ${account.email}`)
            .addFields(
                { name '🔐 CÓDIGO 2FA', value ````n${newCode}n````, inline false },
                { name '📧 Email', value account.email, inline true },
                { name '🔑 Senha', value '' + account.senha + '', inline true },
                { name '⏱️ Expira em', value `${newTimeLeft} segundos`, inline true },
                { name '🎮 Plataforma', value 'Rockstar Games (GTA V, RDR2)', inline false }
            )
            .setFooter({ text 'Use este código para fazer login na Rockstar Launcher  Gg Community © Criado por Miguel®' })
            .setTimestamp();
        
        await msg.edit({ embeds [updatedEmbed], components [row] });
        database.logGeneratedCode(account.id, account.email, newCode);
    }, timeLeft  1000);
}

 Remover conta
async function removeAccount(message, args) {
    const email = args.join(' ');
    
    if (!email) {
        return message.reply('❌ Use `!remover email`nExemplo `!remover miguel@gmail.com`');
    }
    
    const removed = database.removeAccountByEmail(message.author.id, email);
    
    if (removed) {
        const embed = new EmbedBuilder()
            .setColor(0x00FF00)
            .setTitle('✅ Conta Removida')
            .setDescription(`Conta ${email} foi removida com sucesso!`)
            .setFooter({ text 'Gg Community © Criado por Miguel®' });
        
        await message.reply({ embeds [embed] });
    } else {
        await message.reply(`❌ Conta com email ${email} não encontrada.`);
    }
}

 Histórico de códigos
async function showHistory(message, args) {
    const email = args.join(' ');
    
    if (!email) {
        return message.reply('❌ Use `!historico email`');
    }
    
    const history = database.getCodeHistory(email, 10);
    
    if (history.length === 0) {
        return message.reply(`📭 Nenhum código gerado para a conta ${email} ainda.`);
    }
    
    const embed = new EmbedBuilder()
        .setColor(0x00FF00)
        .setTitle(`📜 Histórico de Códigos - ${email}`)
        .setDescription(`Últimos ${history.length} códigos gerados`)
        .setFooter({ text 'Gg Community © Criado por Miguel®' });
    
    history.forEach((item, index) = {
        embed.addFields({
            name `${index + 1}. Código ${item.code}`,
            value `📅 ${new Date(item.generated_at).toLocaleString()}  ${item.used  '✅ Usado'  '⏳ Não usado'}`,
            inline false
        });
    });
    
    await message.reply({ embeds [embed] });
}

async function handleButtonInteraction(interaction) {

    if (customId.startsWith('copy_code_')) {
        const parts = customId.split('_');
        const code = parts[2];
        const email = parts.slice(3).join('_');
        
        await interaction.reply({ 
            content `📋 Código `${code}` copiado para sua área de transferência!`,
            ephemeral true 
        });
        
    } else if (customId.startsWith('refresh_code_')) {
        const parts = customId.split('_');
        const accountId = parseInt(parts[2]);
        const email = parts.slice(3).join('_');
        
        const account = database.getAccountById(accountId, interaction.user.id);
        
        if (!account) {
            return interaction.reply({ content '❌ Conta não encontrada.', ephemeral true });
        }
        
        const code = speakeasy.totp({
            secret account.secret_2fa,
            encoding 'base32',
            step 30,
            digits 6
        });
        
        const now = Math.floor(Date.now()  1000);
        const timeLeft = 30 - (now % 30);
        
        const embed = new EmbedBuilder()
            .setColor(0x00FF00)
            .setTitle('🎮 ROCKSTAR 2FA - Código Gerado')
            .setDescription(`Conta ${account.email}`)
            .addFields(
                { name '🔐 CÓDIGO 2FA', value ````n${code}n````, inline false },
                { name '📧 Email', value account.email, inline true },
                { name '🔑 Senha', value '' + account.senha + '', inline true },
                { name '⏱️ Expira em', value `${timeLeft} segundos`, inline true }
            )
            .setFooter({ text 'Gg Community © Criado por Miguel®' });
        
        const row = new ActionRowBuilder()
            .addComponents(
                new ButtonBuilder()
                    .setCustomId(`copy_code_${code}_${account.email}`)
                    .setLabel('📋 Copiar Código')
                    .setStyle(ButtonStyle.Primary),
                new ButtonBuilder()
                    .setCustomId(`refresh_code_${account.id}_${account.email}`)
                    .setLabel('🔄 Gerar Novamente')
                    .setStyle(ButtonStyle.Success),
                new ButtonBuilder()
                    .setCustomId(`copy_creds_${account.id}_${account.email}`)
                    .setLabel('📧 Copiar EmailSenha')
                    .set
